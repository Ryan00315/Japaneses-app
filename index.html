// --- 🚀 最終強化版翻譯引擎 (不准動到介面) --- window.autoTranslate = async () => { const w = document.getElementById('word').value.trim(); if (!w) return; window.showToast("⏳ AI 正在翻譯..."); try { const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${KEY}`; // 強化指令：要求 AI 必須回傳固定格式，避免 JSON 報錯 const prompt = `Translate the Japanese word "${w}". Return only this format: K:Kanji/M:Meaning Example: K:誕生日/M:生日 Date/Time use numbers. No other words.`; const res = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] }) }); const data = await res.json(); const aiResponse = data.candidates[0].content.parts[0].text; // 🧠 強效掃描器：直接抓取 K: 與 /M: 之間的文字 const kMatch = aiResponse.match(/K:(.*?)\/M:/); const mMatch = aiResponse.match(/\/M:(.*)/); if (kMatch && mMatch) { const kanjiResult = kMatch[1].trim(); const meaningResult = mMatch[1].trim(); // 執行填寫 document.getElementById('reading').value = kanjiResult; document.getElementById('meaning').value = meaningResult; window.showToast("✅ 翻譯成功"); } else { // 如果 AI 回傳格式歪掉，嘗試簡易分割 const fallback = aiResponse.split('/'); if (fallback.length >= 2) { document.getElementById('reading').value = fallback[0].replace('K:', '').trim(); document.getElementById('meaning').value = fallback[1].replace('M:', '').trim(); window.showToast("✅ 翻譯成功 (備援模式)"); } else { throw new Error("Format Mismatch"); } } } catch (e) { console.error("AI 翻譯錯誤:", e); window.showToast("❌ 翻譯失敗"); } };
